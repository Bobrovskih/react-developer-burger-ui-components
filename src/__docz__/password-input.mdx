# PasswordInput


## Basic usage

<Playground>
    {() => {
        const [value, setValue] = React.useState('password');
        const onChange = (e) => {
            setValue(e.target.value);
        };
        return (
            <PasswordInput
                onChange={onChange}
                value={value}
                name={'password'}
            />
        );
   }}
</Playground>
